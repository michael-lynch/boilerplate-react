import Dialog from 'components/ui/Dialog/Dialog';

// forms
import FormLogin from 'components/forms/FormLogin/FormLogin';

// state
const [isDialogOpen, setIsDialogOpen] = useState(false);

  handleClose={() => setIsDialogOpen(false)}

  title="Dialog Title"

<Dialog

      onClick: () => {

      },
    },
  ]}

  mods={['sm']}
